// Program C++ pentru gasirea celei mai mare cifre dintr-un numarului
#include <iostream>
using namespace std;

int main()
{
    int n, max;
    cout<<"Introduceti numarul : ";
    cin>>n;
    // max este initializat cu 0
    max = 0;
    // parcurgerea numarului cifra cu cifra
    while(n != 0)
    {
        // verificam daca ultima cifra a numarului este cea mai mare cifra 
        if(n%10 > max)
            max = n%10;
        // eliminam ultima cifra din numar
        n = n/10;
    }
    cout<<"Cea mai mare cifra a numarului este "<<max;
}
