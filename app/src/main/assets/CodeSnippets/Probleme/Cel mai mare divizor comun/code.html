// Program C++ pentru gasirea celui mai mare divizor comun
#include <iostream>
using namespace std;

int main()
{
    int n1, n2;
    // citim numere de la tastatura
    cout<<"Introduceti numerele cele doua numere: "<<endl;
    cin>>n1>>n2;
    // prin scaderi repetate se ajunge la cel mai mare divizor comun
    while(n1 != n2)
        {
            if(n1 > n2)
                n1 = n1 - n2;
            else
                n2 = n2 - n1;
        }
    cout<<"CMMDC = "<<n1;
    return 0;
}
