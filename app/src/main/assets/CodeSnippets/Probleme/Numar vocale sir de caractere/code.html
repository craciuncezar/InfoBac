// Program C++ pentru calcularea numarului de vocale dintr-un text
#include <iostream>
#include <string.h>
using namespace std;

int main(){
    // citirea textului de la tastatura
    char a[51];
    cin.get(a,50);
    // sir de caractere continand vocalele
    char vocale[]="aeiou";
    int i,nrVocale = 0;
    // parcurgem sirul
    for (i = 0; i < strlen(a); i++){
        // daca sirul cu vocale contine caracterul actual
        // incrementam numarul de vocale
        if (strchr(vocale,a[i]))
            nrVocale++;
    }
    
    cout<<"Numarul de vocale din textul este: "<<nrVocale<<'\n';
    return 0;
}
