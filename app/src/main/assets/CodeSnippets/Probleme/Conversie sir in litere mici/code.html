// Program C++ pentru transformarea literelor unui text in litere mici
#include <iostream>
#include <string.h>
using namespace std;

int main(){
    char a[51];
    cout<<"Introdu textul:"<<endl;
    cin.get(a,50);

    int i;
    // parcurgerea sirul
    for (i = 0; i < strlen(a); i++){
        // verificare daca avem caracter litera mare
        if (a[i] >= 'A' && a[i] <= 'Z')
            a[i] = tolower(a[i]); // transformare in litera mica
    }

    cout<<a;
    return 0;
}
