// Program C++ pentru verificarea unui text palindrom
#include <iostream>
#include <string.h>
using namespace std;

int main(){
// citirea textului
char a[201];
cout<<"Introdu textul:"<<endl;
cin.get(a,200);

int i,ok = 0;

// eliminam " ,!." din sir
for (i = 0; i < strlen(a); i++){
    if (strchr(" ,!.",a[i])){
        int j;
        for (j = i; j < strlen(a); j++){
            a[j] = a[j+1];
        }
    }
}

// comparam caracterele (primul caracter cu ultimul, al 2-lea cu penultimul s.a.m.d)
for (i = 0; i < strlen(a)/2 && ok == 0; i++){
    // folosim tolower pentru a converti literele in litera mica
    // astfel tolower("A") va fi egal cu tolower("a")
    if (tolower(a[i]) != tolower(a[strlen(a) - i - 1]))
        ok = 1;
}

if (ok == 0)
    cout<<"Textul este palindrom\n";
else
    cout<<"Textul nu este palindrom\n";

return 0;
}
