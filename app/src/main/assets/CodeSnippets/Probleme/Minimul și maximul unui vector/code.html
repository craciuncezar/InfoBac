// Program C++ pentru gasirea celui mai mic/mare numar al unui vector
#include <iostream>
using namespace std;

int main(){
    // numarul de elemente
    int n;
    cout<<"Introduceti numarul de elemente"<<endl;
    cin>>n;

    int v[n];
    int i,j;

    // citim elementele vectorului
    cout<<"Introduceti elementele vectorului"<<endl;
    for (i = 0; i < n; i++)
            cin>>v[i];

    int max, min;
    // initializam maximul si minimul cu prima cifra a vectorului
    max = v[0];
    min = v[0];
    // parcurgem vectorul comparand fiecare cifra cu maximul si minimul
    for (i = 1; i < n; i++){
        if(v[i] > max)
            max = v[i];
        else if (v[i] < min)
            min = v[i];
    }
    
    cout<<"Cel mai mic element este "<<min<<", iar cel mai mare element este "<<max;
}
