// Program C++ pentru calcularea sumei cifrelor unui numar
#include <iostream>
using namespace std;

int main()
{
    int n, suma;
    cout<<"Introduceti numarul : ";
    cin>>n;
    // suma este initializata cu 0
    suma = 0;
    // parcurgerea numarului cifra cu cifra
    while(n != 0)
    {
        // adaugam ultima cifra a numarului la suma
        suma = suma + n%10;
        // eliminam ultima cifra din numar
        n = n/10;
    }
    cout<<"Suma cifrelor numarului introdus este "<<suma;
}
