// Program C++ pentru inversarea cifrelor unui numar
#include <iostream>
using namespace std;

int main()
{
int n, invers;
cout<<"Introduceti numarul: ";
cin >> n;
// se construieste inversul numarului
while (n != 0)
    {
        invers = invers * 10 + n % 10; // se inverseaza cifra cu cifra numarul
        n = n / 10; // se taie pe rand cifrele numarului
    }
cout << "Inversul numarului 'n' este: "<< invers;
return 0;
}
