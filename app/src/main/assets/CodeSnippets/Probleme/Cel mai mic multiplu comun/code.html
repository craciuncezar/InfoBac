// Program C++ pentru gasirea celui mai mic multiplu comun
#include <iostream>
using namespace std;

int main()
{
    int n1, n2, max;
    // citim numerele de la tastatura
    cout << "Introdu cele doua numere: ";
    cin >> n1 >> n2;
    
    // salvam numarul mai mare in variabila max
    if(n1 > n2)
        max = n1;
    else
        max = n2;

    do
    {
        if (max % n1 == 0 && max % n2 == 0)
        {
            cout << "CMMMC = " << max;
            break;
        }
        else
            ++max;
    } while (true);
    
    return 0;
}
